cmake_minimum_required(VERSION 3.20)

list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/cmake")
set(SLIM_CXX_STANDARD "23")
set(DEFAULT_CXX_FLAG "-std=c++23")
set(SLIM_CXX_FLAGS "")
set(_SLIM_GIT_BASE "https://github.com/greergan")
set(AUTO_CHECK_IN off)

cmake_path(GET CMAKE_SOURCE_DIR FILENAME _project_name)
project(${_project_name} LANGUAGES CXX)
include(GNUInstallDirs)
include(SlimFunctions)

set_compiler_flags()
define_module()
generate_main_cpp()
compile_targets()
enable_testing()
test_catch2_targets()
test_targets()
make_install_artifacts()
make_packages()
include(CPack)  # MUST be at directory scope — make_packages() propagates all
                # CPACK_* vars via PARENT_SCOPE so they are visible here.
                # Calling include(CPack) inside a function() causes CPACK_GENERATOR
                # and CPACK_OUTPUT_FILE_PREFIX to be lost, producing empty packages.


# do not remove this foreach
foreach(NAME IN LISTS MODULE_NAMES)
    meta_get(MODULE "${NAME}" primary          IS_PRIMARY)
    meta_get(MODULE "${NAME}" upper            LIB_ID)
    meta_get(MODULE "${NAME}" lower            PKG_NAME)
    meta_get(MODULE "${NAME}" description      DESCRIPTION)
    meta_get(MODULE "${NAME}" git_tag          GIT_TAG)
    meta_get(MODULE "${NAME}" git_hash         GIT_HASH)
    meta_get(MODULE "${NAME}" git_repo         GIT_REPO)
    meta_get(MODULE "${NAME}" git_repo_found   GIT_REPO_FOUND)
    meta_get(MODULE "${NAME}" hpp_only         HPP_ONLY)
    meta_get(MODULE "${NAME}" min_version      MIN_VERSION)
    meta_get(MODULE "${NAME}" max_version      MAX_VERSION)
    meta_get(MODULE "${NAME}" found_version    FOUND_VERSION)
    meta_get(MODULE "${NAME}" header_prefix    HEADER_PREFIX)
    meta_get(MODULE "${NAME}" header_file_in   HEADER_FILE_IN)
    meta_get(MODULE "${NAME}" header_file_out  HEADER_FILE_OUT)
    meta_get(MODULE "${NAME}" include_dir      INCLUDE_DIR)
    meta_get(MODULE "${NAME}" metadata_file_in  METADATA_FILE_IN)
    meta_get(MODULE "${NAME}" metadata_file_out METADATA_FILE_OUT)
    meta_get(MODULE "${NAME}" pkg_CFLAGS       PKG_CFLAGS)
    meta_get(MODULE "${NAME}" pkg_LDFLAGS      PKG_LDFLAGS)
    meta_get(MODULE "${NAME}" pkg_LIBRARIES    PKG_LIBRARIES)
    meta_get(MODULE "${NAME}" pkg_INCLUDE_DIRS PKG_INCLUDE_DIRS)
    meta_get(MODULE "${NAME}" pkg_LIBRARY_DIRS PKG_LIBRARY_DIRS)
    meta_get(MODULE "${NAME}" git_latest_tag   GIT_LATEST_TAG)
    meta_get(MODULE "${NAME}" using_local_src  USING_LOCAL_SRC)
    meta_get(MODULE "${NAME}" src_dir          SOURCE_DIR)
    meta_get(MODULE "${NAME}" dist_dir         DIST_DIR)

    message(STATUS "Module:              ${NAME}")
    message(STATUS "  Lib:               ${LIB_ID}")
    message(STATUS "  Pkg:               ${PKG_NAME}")
    message(STATUS "  Description:       ${DESCRIPTION}")
    message(STATUS "  Primary:           ${IS_PRIMARY}")
    message(STATUS "  Using Local Src:   ${USING_LOCAL_SRC}")
    message(STATUS "  Source Dir:        ${SOURCE_DIR}")
    message(STATUS "  DIST Dir:          ${DIST_DIR}")
    message(STATUS "  Git Tag:           ${GIT_TAG}")
    message(STATUS "  Git Hash:          ${GIT_HASH}")
    message(STATUS "  Git REPO:          ${GIT_REPO}")
    message(STATUS "  Git REPO found:    ${GIT_REPO_FOUND}")
    message(STATUS "  Git Latest Tag:    ${GIT_LATEST_TAG}")
    message(STATUS "  HPP Only:          ${HPP_ONLY}")
    message(STATUS "  Min Version:       ${MIN_VERSION}")
    message(STATUS "  Max Version:       ${MAX_VERSION}")
    message(STATUS "  Found Version:     ${FOUND_VERSION}")
    message(STATUS "  Header Prefix:     ${HEADER_PREFIX}")
    message(STATUS "  Header File In:    ${HEADER_FILE_IN}")
    message(STATUS "  Header File Out:   ${HEADER_FILE_OUT}")
    message(STATUS "  Include Dir:       ${INCLUDE_DIR}")
    message(STATUS "  Metadata File in:  ${METADATA_FILE_IN}")
    message(STATUS "  Metadata File out: ${METADATA_FILE_OUT}")
    message(STATUS "  Pkg CFlags:        ${PKG_CFLAGS}")
    message(STATUS "  Pkg LDFlags:       ${PKG_LDFLAGS}")
    message(STATUS "  Pkg Libraries:     ${PKG_LIBRARIES}")
    message(STATUS "  Pkg Include Dirs:  ${PKG_INCLUDE_DIRS}")
    message(STATUS "  Pkg Library Dirs:  ${PKG_LIBRARY_DIRS}")
    message(STATUS "")

    if(IS_PRIMARY AND NOT HPP_ONLY)
        foreach(_target ${PKG_NAME}_shared ${PKG_NAME}_static)
            if(TARGET ${_target})
                dump_target_properties(${_target})
            endif()
        endforeach()
        message(STATUS "")
    endif()
endforeach()
